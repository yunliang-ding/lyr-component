## 基本使用

```tsx
/**
 * background: '#f6f7f9'
 */
import React from 'react';
import { DragList } from 'lyr-design';

export default () => {
  return (
    <div style={{ width: 100 }}>
      <DragList
        defaultActiveKey={3}
        onClick={(item) => {
          console.log(item);
        }}
        onChange={(list) => {
          console.log(list);
        }}
        list={[1, 2, 3, 4, 5, 6, 7, 8, 9].map((i) => {
          return {
            key: i,
            label: <div style={{ padding: 6 }}>{i}</div>,
          };
        })}
      />
    </div>
  );
};
```
